8def mask_account_card(text_data: str) -> str: 

9 """ 

10 Функция принимает любую строку, в которой имеется 16тизначный номер карты 

11 или 20ти значным номер счёта и определяет куда дальше направлять 

12 После первого найденного номера обрабатывает его и возвращает текст до номера 

13 плюс замаскированный номер — без остальной части строки 

14 """ 

15 card_found = bool(re.search(r"\b(\d{16})\b", text_data)) 

16 account_found = bool(re.search(r"\b(\d{20})\b", text_data)) 

17 

18 if card_found and account_found: 

19 return "error: invalid format" 

20 if not card_found and not account_found: 

21 return "" 

22 

23 # ищем 16ти значный номер карты в строке 

24 card_match = re.search(r"\b(\d{16})\b", text_data) 

25 if card_match: 

26 card_number = card_match.group(1) 

27 masked_card = get_mask_card_number(card_number) 

28 start, end = card_match.span() 

29 return text_data[:start] + masked_card 

30 

31 # ищем 20ти значный номер счёта в строке 

32 account_match = re.search(r"\b(\d{20})\b", text_data) 

33 if account_match: 

34 account_number = account_match.group(1) 

35 masked_account = get_mask_account(account_number) 

36 start, end = account_match.span() 

37 return text_data[:start] + masked_account 

38 

39 return ""
